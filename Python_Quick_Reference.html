Classes (OOP Basics)
Defining a class
class Animal: # class attribute (shared by all instances) species = "Unknown" # __init__ = constructor def __init__(self, name, age): self.name = name # instance attribute self.age = age # method def speak(self): return f"{self.name} speaks" # __str__ — string representation def __str__(self): return f"Animal({self.name})" # create instance dog = Animal("Rex", 5) dog.speak() # "Rex speaks" print(dog) # "Animal(Rex)"
Inheritance
class Dog(Animal): # inherits Animal species = "Canine" def __init__(self, name, age, breed): super().__init__(name, age) # call parent self.breed = breed # override parent method def speak(self): return f"{self.name} says Woof!" d = Dog("Rex", 5, "Lab") d.speak() # "Rex says Woof!" isinstance(d, Animal) # True
Scapy
Read & iterate packets
from scapy.all import rdpcap, wrpcap, sniff # read pcap file pkts = rdpcap("capture.pcap") # iterate all packets for pkt in pkts: print(pkt.summary()) # check if packet has a layer if pkt.haslayer("TCP"): print(pkt["TCP"].dport) # access specific packet and layer field pkts[0]["IP"].src # source IP of packet 0 pkts[0]["TCP"].flags # TCP flags pkts[0]["DNS"].qd.qname # DNS query name # write packets to pcap wrpcap("output.pcap", pkts)
Craft & send packets
from scapy.all import Ether, IP, TCP, UDP, send, sendp # craft an IP/TCP packet pkt = IP(dst="192.168.1.1") / TCP(dport=80) # add Ethernet layer pkt = Ether() / IP(dst="10.0.0.1") / TCP() # send at L3 (IP layer) send(pkt) # send at L2 (Ethernet layer) sendp(pkt, iface="eth0") # layer names (case-sensitive!) # Ether · IP · TCP · UDP · DNS · RAW # sniff live traffic pkts = sniff(count=10, filter="tcp")
Sockets
TCP client & server
import socket # ── TCP CLIENT ── s = socket.socket(socket.AF_INET, # IPv4 socket.SOCK_STREAM) # TCP s.connect(("192.168.1.1", 4444)) s.send(b"hello") data = s.recv(1024) # receive up to 1024 bytes s.close() # ── TCP SERVER ── srv = socket.socket(socket.AF_INET, socket.SOCK_STREAM) srv.bind(("", 4444)) # bind to all interfaces srv.listen(1) # queue 1 connection conn, addr = srv.accept() # wait for client print(f"Connected: {addr}") conn.close() srv.close()
UDP & IPv6 sockets
import socket # ── UDP (connectionless) ── s = socket.socket(socket.AF_INET, # IPv4 socket.SOCK_DGRAM) # UDP s.sendto(b"data", ("8.8.8.8", 53)) data, addr = s.recvfrom(512) # ── IPv6 TCP ── s6 = socket.socket(socket.AF_INET6, # IPv6 socket.SOCK_STREAM) # TCP s6.connect(("::1", 8080)) # ── error handling ── try: s.connect(("host", 9999)) except socket.error as e: print(f"Socket error: {e}")
Subprocess
Popen, run, call
import subprocess # run — simple, returns CompletedProcess r = subprocess.run( ["ls", "-la"], capture_output=True, text=True # output as str not bytes ) print(r.stdout) print(r.returncode) # 0 = success # call — run and wait, returns exit code code = subprocess.call(["ping", "-c1", "8.8.8.8"]) # Popen — more control, non-blocking proc = subprocess.Popen( ["nmap", "-sV", "target"], stdout=subprocess.PIPE, stderr=subprocess.PIPE, text=True ) out, err = proc.communicate() # wait & get output
